var userCtrl = require('../controllers/user');
var msgCtrl = require('../controllers/message');
var utils = require('../utils');
var stopwordService = utils.stopwordService;
var userModel = require('../models/user');
var globalVal = require("../global");
 
var search = function(token, type, fromIndex, toIndex, content, callback) {
    userModel.validate({token:token}, function(user) {
        if (user) {
            if (type == 1 || type == 0) {
                if (fromIndex) {
                    callback({Error: {code: globalVal.notfound}});
                    return;
                }
                if (toIndex) {
                    callback({Error: {code: globalVal.notfound}});
                    return;
                }
                userCtrl.searchForUserLists(type, content, function (res) {
                    callback(res);
                });
                return;
            }
            if (type == 2 || type == 3 || type == 4) {
                stopwordService.isValid(content, function (res) {
                    if (res) {
                        if (!fromIndex) {
                            callback({Error: {code: globalVal.notfound}});
                            return;
                        }
                        if (!toIndex) {
                            callback({Error: {code: globalVal.notfound}});
                            return;
                        }
                        if(type == 2 || type == 3) {
                            msgCtrl.searchForMessages(type, content, fromIndex, toIndex, function (res) {
                                callback(res);
                            });
                        }
                        else{
                            msgCtrl.searchForPrivateMessages(token,content, fromIndex, toIndex,function (res){
                                callback(res);
                            });
                        }
                    }
                    else callback({Error: {code: globalVal.forbidden}});
 
                });
                return;
            }
            return callback({Error: {code: globalVal.notfound}});
        }
        else callback({Error: {code: globalVal.unauthorized}});
    });
};
 
exports.search = search;
